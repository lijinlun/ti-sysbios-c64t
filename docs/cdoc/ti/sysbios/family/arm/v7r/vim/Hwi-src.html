    44    /*!
    45     * TODO add examples showing how to change stack address
    46     *  ======== Hwi ========
    47     *  Hardware Interrupt Support Module.
    48     *
    49     *  This Hwi module provides Hercules VIM specific implementations of the
    50     *  APIs defined in {@link ti.sysbios.interfaces.IHwi IHwi}.
    51     *
    52     *  Additional ARM device-specific APIs are also provided.
    53     *
    54     *  @a(Minimal Latency Interrupts)
    55     *  For applications requiring extremely low interrupt latency, this Hwi module
    56     *  allows the user to create FIQ interrupts that bypass the SYS/BIOS interrupt
    57     *  dispatcher. Though not a precisely correct classification, these interrupts
    58     *  are referred to as "Zero latency" interrupts.
    59     *
    60     *  "FIQ" aka "Zero latency" interrupts can be created by setting the Hwi_Param
    61     *  {@link #type} to Hwi_Type_FIQ. FIQ interrupts offer low interrupt latency
    62     *  as they do not have to pass through the regular SYS/BIOS interrupt
    63     *  dispatcher and are always enabled. When auto nesting is enabled and masking
    64     *  option ALL or LOWER is used, some or all of the VIM interrupt channels will
    65     *  be disabled while the interrupt is being serviced. However, none of the
    66     *  channels corresponding to "Zero latency" interrupts are disabled (masked).
    67     *
    68     *  The ISR handler function for a "Zero latency" interrupt must use the
    69     *  "interrupt" keyword in the function definition and should have no
    70     *  parameters. Using the "interrupt" keyword will ensure that the necessary
    71     *  registers are saved on entry into the interrupt routine and are restored
    72     *  upon exit from the interrupt routine.
    73     *
    74     *  Since the function pointer passed to Hwi_create()/Hwi_construct() has a
    75     *  different function signature (i.e. it accepts an argument), the
    76     *  "Zero latency" function must be type-casted to {@link #FuncPtr Hwi_FuncPtr}
    77     *  type before being passed to Hwi_create()/Hwi_construct() in order to avoid
    78     *  any compiler warnings.
    79     *
    80     *  Unlike regular IRQ interrupts, FIQ interrupts do not run on the System stack
    81     *  but on their own FIQ stack. The stack pointer, size and section name for the
    82     *  FIQ stack can be set using the {@link #fiqStack}, {@link #fiqStackSize}
    83     *  and {@link #fiqStackSection} module wide configuration params.
    84     *
    85     *  @a(Constraints of using "FIQ" aka "Zero latency" interrupts)
    86     *  Interrupts configured to bypass the dispatcher are not allowed
    87     *  to call ANY SYS/BIOS APIs that effect thread scheduling. Examples
    88     *  of API that should no be invoked are:
    89     *
    90     *  @p(dlist)
    91     *    - Swi_post(),
    92     *    - Semaphore_post(),
    93     *    - Event_post(),
    94     *    - Task_yield()
    95     *  @p
    96     *
    97     *  Here's an example showing how to create a Hwi of FIQ type:
    98     *  @p(code)
    99     *  *.cfg:
   100     *  var Hwi = xdc.useModule('ti.sysbios.family.arm.v7r.vim.Hwi');
   101     *  Hwi.fiqStackSize = 2048;
   102     *  Hwi.fiqStackSection = ".myFiqStack"
   103     *  Program.sectMap[".myFiqStack"] = "RAM";
   104     *
   105     *  *.c:
   106     *  #include <xdc/std.h>
   107     *  #include <xdc/runtime/System.h>
   108     *
   109     *  #include <ti/sysbios/BIOS.h>
   110     *  #include <ti/sysbios/family/arm/v7r/vim/Hwi.h>
   111     *
   112     *  #include <xdc/cfg/global.h>
   113     *
   114     *  Void interrupt myIsrFIQ()
   115     *  {
   116     *      ...
   117     *  }
   118     *
   119     *  Void main(Void)
   120     *  {
   121     *      Hwi_Params hwiParams;
   122     *
   123     *      Hwi_Params_init(&hwiParams);
   124     *      hwiParams.type = Hwi_Type_FIQ;
   125     *      Hwi_create(INT_NUM_FIQ, (Hwi_FuncPtr)myIsrFIQ, &hwiParams, NULL);
   126     *      ...
   127     *
   128     *      BIOS_start();
   129     *  }
   130     *  @p
   131     *
   132     *  @a(Interrupt Channel configuration)
   133     *  Each VIM interrupt request (source) can be mapped to any of the interrupt
   134     *  channels. Lower numbered channels in each FIQ and IRQ have higher priority.
   135     *  Therefore, channel mapping provides a mechanism for prioritizing the
   136     *  interrupt requests.
   137     *
   138     *  Additionally, it is possible to configure a channel interrupt as a wakeup
   139     *  interrupt so it can bring the core out of low power mode (LPM).
   140     *
   141     *  This module has a {@link #configChannelMeta Hwi.configChannelMeta()}
   142     *  function that can be invoked from the cfg script to statically
   143     *  (at build time) change the channel mapping and wakeup functionality of a
   144     *  channel interrupt.
   145     *
   146     *  By default, all interrupt request sources are direct mapped to channels
   147     *  (i.e. interrupt request N is mapped to channel N) and wakeup feature is
   148     *  enabled for all interrupts.
   149     *
   150     *  Here's an example showing how to use this function:
   151     *  @p(code)
   152     *  *.cfg:
   153     *  var Hwi = xdc.useModule('ti.sysbios.family.arm.v7r.vim.Hwi');
   154     *
   155     *  // Map interrupt request line 86 to channel 2 and disable wakeup feature
   156     *  Hwi.configChannelMeta(2, 86, false);
   157     *  @p
   158     *
   159     *  @a(Non-dispatched interrupts)
   160     *  Cortex-R5 supports hardware vectored IRQ interrupts (i.e. interrupts are
   161     *  automatically dispatched to ISR). In order to leverage this feature for
   162     *  a given IRQ interrupt, the SYS/BIOS Hwi dispatcher needs to be bypassed
   163     *  for the particular interrupt. The Hwi params contain a
   164     *  {@link #useDispatcher} param that can be used for this purpose.
   165     *
   166     *  When {@link #useDispatcher} Hwi param is set to false, the Hwi function's
   167     *  address is written to the hardware interrupt vector table (VIM RAM). After
   168     *  an IRQ interrupt is received by the CPU, the CPU reads the address of the
   169     *  ISR directly from the VIM RAM and jumps to the function (See
   170     *  {@link #useDispatcher} for more info).
   171     *
   172     *  @p(html)
   173     *  <h3> Calling Context </h3>
   174     *  <table border="1" cellpadding="3">
   175     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center"></colgroup>
   176     *
   177     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
   178     *    <!--                                                                                                                 -->
   179     *    <tr><td> {@link #clearInterrupt}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   180     *    <tr><td> {@link #create}           </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   181     *    <tr><td> {@link #disable}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   182     *    <tr><td> {@link #disableInterrupt} </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   183     *    <tr><td> {@link #disableIRQ}       </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   184     *    <tr><td> {@link #enable}           </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   185     *    <tr><td> {@link #enableInterrupt}  </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   186     *    <tr><td> {@link #enableIRQ}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   187     *    <tr><td> {@link #getHandle}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   188     *    <tr><td> {@link #Params_init}      </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   189     *    <tr><td> {@link #restore}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   190     *    <tr><td> {@link #restoreInterrupt} </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   191     *    <tr><td> {@link #restoreIRQ}       </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   192     *    <tr><td> {@link #construct}        </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   193     *    <tr><td> {@link #delete}           </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   194     *    <tr><td> {@link #destruct}         </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   195     *    <tr><td> {@link #getHookContext}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   196     *    <tr><td> {@link #reconfig}         </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   197     *    <tr><td> {@link #setFunc}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   198     *    <tr><td> {@link #setHookContext}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   199     *    <tr><td colspan="6"> Definitions: <br />
   200     *       <ul>
   201     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
   202     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
   203     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
   204     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
   205     *           <ul>
   206     *             <li> In your module startup after this module is started (e.g. Hwi_Module_startupDone() returns TRUE). </li>
   207     *             <li> During xdc.runtime.Startup.lastFxns. </li>
   208     *             <li> During main().</li>
   209     *             <li> During BIOS.startupFxns.</li>
   210     *           </ul>
   211     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
   212     *           <ul>
   213     *             <li> During xdc.runtime.Startup.firstFxns.</li>
   214     *             <li> In your module startup before this module is started (e.g. Hwi_Module_startupDone() returns FALSE).</li>
   215     *           </ul>
   216     *       </ul>
   217     *    </td></tr>
   218     *
   219     *

   329        /*!
   330         *  ======== core0VectorTableAddress ========
   331         *  Determines the location of Core0's Interrupt Vector Table on a
   332         *  Dual-Core device. Default is device dependent.
   333         *
   334         *  On Dual-Core devices, both Cortex-R5 cores share a common reset
   335         *  vector table. In order to allow the 2 cores to register their own
   336         *  exception handlers, each core generates its own clone of the reset
   337         *  vector table and initializes it with its own exception handler
   338         *  addresses. The core specific vector tables are placed at fixed
   339         *  addresses so that the exception handler functions called by the
   340         *  common reset vector table known each core's vector table address
   341         *  and are able to reference it once they detect which core the
   342         *  application is currently running on.
   343         *
   344         *  The address of Core0's vector table is determined by this parameter.
   345         *
   346         *  Here are the default Core0 vector table addresses for all supported
   347         *  Dual-Core devices:
   348         *  @p(code)
   349         *   ----------------------------------------------------
   350         *  | Device name | Core0's default vector table address |
   351         *   ----------------------------------------------------
   352         *  | RM57D8xx    | 0x100                                |
   353         *   ----------------------------------------------------
   354         *  @p
   355         *
   356         *  @a(Note)
   357         *  If changing Core0's vector table address, it is not necessary to
   358         *  rebuild Core1's application as it does not need to know the location
   359         *  of Core0's vector table.
   360         */
   361        config Ptr core0VectorTableAddress;
   362    
   363        /*!
   364         *  ======== core1VectorTableAddress ========
   365         *  Determines the location of Core1's Interrupt Vector Table on a
   366         *  Dual-Core device. Default is device dependent.
   367         *
   368         *  On Dual-Core devices, both Cortex-R5 cores share a common reset
   369         *  vector table. In order to allow the 2 cores to register their own
   370         *  exception handlers, each core generates its own clone of the reset
   371         *  vector table and initializes it with its own exception handler
   372         *  addresses. The core specific vector tables are placed at fixed
   373         *  addresses so that the exception handler functions called by the
   374         *  common reset vector table known each core's vector table address
   375         *  and are able to reference it once they detect which core the
   376         *  application is currently running on.
   377         *
   378         *  Core0's vector table is always placed at 0x100 while the address of
   379         *  Core1's vector table is determined by this parameter.
   380         *
   381         *  Here are the default Core1 vector table addresses for all supported
   382         *  Dual-Core devices:
   383         *  @p(code)
   384         *   ----------------------------------------------------
   385         *  | Device name | Core1's default vector table address |
   386         *   ----------------------------------------------------
   387         *  | RM57D8xx    | 0x200000                             |
   388         *   ----------------------------------------------------
   389         *  @p
   390         *
   391         *  @a(Note)
   392         *  If changing Core1's vector table address, it is important to rebuild
   393         *  Core0's application with the same change as Core0 owns the common
   394         *  reset vector table and the common exception handler functions need
   395         *  to know Core1's vector table address so they can determine the
   396         *  address of the handler function they need to jump to.
   397         */
   398        config Ptr core1VectorTableAddress;

   618         *  Use the SYS/BIOS Hwi dispatcher. Default is true.
   619         *
   620         *  This param can be set to false for IRQ interrupts to bypass
   621         *  the SYS/BIOS interrupt dispatcher. FIQ interrupts do not go
   622         *  through the dispatcher and therefore this param has no affect
   623         *  on FIQ interrupts.
   624         *
   625         *  Unlike dispatched IRQ interrupts, non-dispatched IRQ interrupts
   626         *  do not use the system stack. The stack pointer, stack size and
   627         *  stack section for the non-dispatched interrupts can be set using
   628         *  the {@link #nonDispatchedIRQStack}, {@link #nonDispatchedIRQStackSize}
   629         *  and {@link #nonDispatchedIRQStackSection} module wise config params.
   630         *
   631         *  @a(constraints)
   632         *  - Interrupts configured to bypass the dispatcher are not allowed
   633         *  to call ANY SYS/BIOS APIs that effect thread scheduling. Examples
   634         *  of API that should not be invoked are:
   635         *
   636         *  @p(dlist)
   637         *    - Swi_post(),
   638         *    - Semaphore_post(),
   639         *    - Event_post(),
   640         *    - Task_yield()
   641         *  @p
   642         *
   643         *  - Additionally, although the signature for a non-dispatched interrupt
   644         *  function is the same as that for a dispatched interrupt
   645         *  (see {@link #FuncPtr}), no argument is actually passed to the
   646         *  non-dispatched ISR handler.
   647         *
   648         *  //TODO we need the "interrupt" keyword
   649         */
   650        config Bool useDispatcher = true;
