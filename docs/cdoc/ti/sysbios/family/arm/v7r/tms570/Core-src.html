    36    import xdc.runtime.Error;
    37    
    38    /*!
    39     *  ======== Core ========
    40     *  Core Identification Module.
    41     *
    42     *  The Core module is used to define which core within a dual core
    43     *  "R5" subsystem an application is being built for.
    44     *
    45     *  At runtime, a comparison is made between the configured Core.id
    46     *  and the value of MPIDR (bits 7:0). If they do not agree, an Error
    47     *  is raised.
    48     *
    49     *  Core 0's vector table is placed at 0x100 and Core 1's vector table
    50     *  is placed at 0x20000.
    51     *
    52     *  @a(Bootup sequence)
    53     *  When each Cortex-R5 core comes out of reset, it runs a routine that
    54     *  determines the source of reset and depending on the source either
    55     *  continues executing the bootup sequence or calls a reset callback
    56     *  function corresponding to the detected reset source.
    57     *
    58     *  In case the reset source is a PowerOn/Debug/External reset signal,
    59     *  the normal bootup sequence is continued (i.e. data section init
    60     *  and execution of various kernel module startup functions before
    61     *  calling main() function).
    62     *
    63     *  For other reset sources, the callback functions corresponding to
    64     *  the reset source are called. Here's an example showing how to register
    65     *  a software reset callback function:
    66     *
    67     *  @p(code)
    68     *  var Core = xdc.useModule('ti.sysbios.family.arm.v7r.tms570.Core');
    69     *  Core.swResetFunc = '&myfunc';
    70     *  @p
    71     */
    72    
    73    @ModuleStartup          /* to configure static timers */
    74    @Template ("./Core.xdt")
    75    
    76    module Core inherits ti.sysbios.interfaces.ICore
    77    {
    78        /*! Reset function type definition. */
    79        typedef Void (*ResetFuncPtr)(void);
    80    
    81        /*!
    82         *  @_nodoc
    83         *  Reset sources
    84         */
    85        enum ResetSource
    86        {
    87            ResetSource_POWERON       =   0x8000U,
    88            ResetSource_OSC_FAILURE   =   0x4000U,
    89            ResetSource_WATCHDOG      =   0x2000U,
    90            ResetSource_WATCHDOG2     =   0x1000U,
    91            ResetSource_DEBUG         =   0x0800U,
    92            ResetSource_INTERCONNECT  =   0x0080U,
    93            ResetSource_CPU1          =   0x0040U,
    94            ResetSource_CPU0          =   0x0020U,
    95            ResetSource_SW            =   0x0010U,
    96            ResetSource_EXT           =   0x0008U,
    97            ResetSource_NO_RESET      =   0x0000U
    98        };
    99    
   100        /*!
   101         *  Error raised if Core.id does not match the contents
   102         *  of MPIDR [7:0] register.
   103         */
   104        config Error.Id E_mismatchedIds = {
   105            msg: "E_mismatchedIds: Core_Id: %d does not match hardware core Id: %d"
   106        };
   107    
   108        override config UInt numCores;
   109    
   110        /*!
   111         *  R5 Core ID, default is Core 0
   112         *
   113         *  Used for making static decisions based on Core ID
   114         */
   115        config UInt id = 0;
   116    
   117        /*!
   118         *  ======== cpu0ResetFunc ========
   119         *  CPU0 reset function pointer
   120         */
   121        metaonly config ResetFuncPtr cpu0ResetFunc = null;
   122    
   123        /*!
   124         *  ======== cpu1ResetFunc ========
   125         *  CPU1 reset function pointer
   126         */
   127        metaonly config ResetFuncPtr cpu1ResetFunc = null;
   128    
   129        /*!
   130         *  ======== interconnectResetFunc ========
   131         *  Interconnect reset function pointer
   132         */
   133        metaonly config ResetFuncPtr interconnectResetFunc = null;
   134    
   135        /*!
   136         *  ======== oscFailureResetFunc ========
   137         *  Oscillator failure reset function pointer
   138         */
   139        metaonly config ResetFuncPtr oscFailureResetFunc = null;
   140    
   141        /*!
   142         *  ======== swResetFunc ========
   143         *  Software reset function pointer
   144         */
   145        metaonly config ResetFuncPtr swResetFunc = null;
   146    
   147        /*!
   148         *  ======== watchdogResetFunc ========
   149         *  Watchdog reset function pointer
   150         */
   151        metaonly config ResetFuncPtr watchdogResetFunc = null;
   152    
   153        /*!
   154         *  ======== watchdog2ResetFunc ========
   155         *  Watchdog2 reset function pointer
   156         */
   157        metaonly config ResetFuncPtr watchdog2ResetFunc = null;
   158    
   159        @Macro
   160        override UInt hwiDisable();
   161    
   162        @Macro
   163        override UInt hwiEnable();
   164    
   165        @Macro
   166        override Void hwiRestore(UInt key);
   167    
   168    internal:
   169    
   170        /*
   171         *  ======== syncStartup ========
   172         *  Sync Core0 and Core1 during startup.
   173         */
   174        Void syncStartup();
   175    
   176        /*
   177         *  ======== getResetSource ========
   178         */
   179        ResetSource getResetSource();
   180    
   181        /*
   182         *  ======== resetC ========
   183         */
   184        Void resetC();
   185    }
