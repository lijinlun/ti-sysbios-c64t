    37    package ti.sysbios.family.arm.msp432.init;
    38    
    39    import xdc.rov.ViewInfo;
    40    import xdc.runtime.Assert;
    41    
    42    /*!
    43     *  ======== Boot ========
    44     *  MSP432 device Boot Support.
    45     *
    46     *  The Boot module supports boot initialization for MSP432 devices.
    47     *  A special boot init function is created based on the configuration
    48     *  settings for this module.  This function is hooked into the
    49     *  xdc.runtime.Reset.fxns[] array and called very early at boot time.
    50     *
    51     *  The code to support the boot module is placed in a separate section
    52     *  named `".text:.bootCodeSection"` to allow placement of this section in
    53     *  the linker .cmd file if necessary. This section is a subsection of the
    54     *  `".text"` section so this code will be placed into the .text section unless
    55     *  explicitly placed, either through
    56     *  `{@link xdc.cfg.Program#sectMap Program.sectMap}` or through a linker
    57     *  command file.
    58     */
    59    @Template("./Boot.xdt")
    60    module Boot
    61    {
    62       /*! clock speed setting */
    63        enum SpeedOpt {
    64            SpeedOpt_Low = 0,
    65            SpeedOpt_Medium = 1,
    66            SpeedOpt_High = 2
    67        };
    68    
    69        metaonly struct ModuleView {
    70            Bool    configureClocks;
    71            Bool    disableWatchdog;
    72        }
    73    
    74        @Facet
    75        metaonly config ViewInfo.Instance rovViewInfo =
    76            ViewInfo.create({
    77                viewMap: [
    78                [
    79                    'Module',
    80                    {
    81                        type: ViewInfo.MODULE,
    82                        viewInitFxn: 'viewInitModule',
    83                        structName: 'ModuleView'
    84                    }
    85                ],
    86                ]
    87            });
    88    
    89        /*!
    90         *  Clock configuration flag, default is true.
    91         *
    92         *  Set to false to disable clock configuration.
    93         *
    94         *  Clock configuration will setup the clock system (CS), VCORE, and
    95         *  Flash wait states appropriately, for one of three different device
    96         *  speed options, as selected by `{@link #speedSelect}`.
    97         */
    98        metaonly config Bool configureClocks = true;
    99    
   100        /*! Clock speed selection, default is SpeedOpt_High.
   101         *
   102         *  This enumeration is used to select one of three different speed options
   103         *  that will be configured when `{@link #configureClocks}` is set to
   104         *  "true".
   105         *
   106         *   SpeedOpt_High will configure:
   107         *      MCLK =   48MHz from DCO
   108         *      HSMCLK = 24MHz from DCO
   109         *      SMCLK =  12MHz from DCO
   110         *      ACLK =   32KHz from REFOCLK
   111         *      BCLK =   32KHz from REFOCLK
   112         *      VCORE = 1 (AM1_LDO mode)
   113         *      Flash BNK0 and BNK1 read wait states = 2
   114         *
   115         *   SpeedOpt_Medium will configure:
   116         *      MCLK =   24MHz from DCO
   117         *      HSMCLK =  6MHz from DCO
   118         *      SMCLK =   6MHz from DCO
   119         *      ACLK =   32KHz from REFOCLK
   120         *      BCLK =   32KHz from REFOCLK
   121         *      VCORE = 1 (AM1_LDO mode)
   122         *      Flash BNK0 and BNK1 read wait states = 1
   123         *
   124         *   SpeedOpt_Low will configure:
   125         *      MCLK =   12MHz from DCO
   126         *      HSMCLK =  3MHz from DCO
   127         *      SMCLK =   3MHz from DCO
   128         *      ACLK =   32KHz from REFOCLK
   129         *      BCLK =   32KHz from REFOCLK
   130         *      VCORE = 0 (AM0_LDO mode)
   131         *      Flash BNK0 and BNK1 read wait states = 0
   132         */
   133        metaonly config SpeedOpt speedSelect = SpeedOpt_High;
   134    
   135        /*!
   136         *  Watchdog disable configuration flag, default is true.
   137         *
   138         *  Set to false to disable the disabling of the watchdog.
   139         */
   140        metaonly config Bool disableWatchdog = true;
   141    
   142        /*!
   143         *  @_nodoc
   144         *  ======== registerFreqListener ========
   145         *  Register a module to be notified whenever the frequency changes.
   146         *
   147         *  The registered module must have a function named 'fireFrequencyUpdate'
   148         *  which takes the new frequency as an argument.
   149         */
   150        function registerFreqListener();
   151    
   152    internal:
   153    
   154        /*
   155         *  ======== init ========
   156         *  Generated entry point for clock and watchdog initialization.
   157         *
   158         *  Installed as a Startup.firstFxn.
   159         */
   160        Void init();
   161    
   162        /*!
   163         *  computed cpu frequency based on clock settings
   164         */
   165        metaonly config UInt computedCpuFrequency;
   166    
   167    };
